```js
    // 删除文件夹
    git rm -rf --cached folderName
    
    // 创建分支
    git branch <name>
    
    // 切换分支，加上 -b 表示创建并切换
    git checkout <name>
    git checkout -b <name>
    
    // 合并分支到当前分支
    git merge <name>
     
    // 提交
    git add -A
    git commit -m '操作说明'
    git push
```
